File文件操作 常见宏

Syntax

MR_IS_FILE   #define MR_IS_FILE 1   文件
MR_IS_DIR   #define MR_IS_DIR 2   目录
MR_IS_INVALID   #define MR_IS_INVALID 8   无效(非文件、非目录)

MR_FILE_RDONLY   #define MR_FILE_RDONLY   1以只读的方式打开文件
MR_FILE_WRONLY   #define MR_FILE_WRONLY   2以只写的方式打开文件
MR_FILE_RDWR   #define MR_FILE_RDWR   4以读写的方式打开文件
MR_FILE_CREATE   #define MR_FILE_CREATE   8如果文件不存在，创建该文件
MR_FILE_RECREATE   #define MR_FILE_RECREATE   16无论文件存不存在，都重新创建该文件
