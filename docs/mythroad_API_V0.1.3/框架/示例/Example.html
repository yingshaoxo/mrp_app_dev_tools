代码取自“泡泡龙”游戏入口部分

   1、mrc_init
      int32 mrc_init(void){
      VariablesInit();//变量初始化
      mrc_initChargeOverSea();//付费模块初始化
      chargeLanguageInit();//付费字符串资源装载
      Mggq_a_Enter_Game();//进入游戏
      }//进入游戏

   2、mrc_pause
      int32 mrc_pause(void)
      {
      Mggq_a_Suspend_Game();
      return MR_SUCCESS;
      }

   3、mrc_resume
      int32 mrc_resume(void)
      {
      Mggq_a_Resume_Game();
      return MR_SUCCESS;
      }

   4、mrc_exitApp
      int32 mrc_exitApp(void)
     {
     Mggq_a_ReleaseAllResource();
     return MR_SUCCESS;
     }

   5、mrc_appEventp
      int32 mrc_appEvent(int32 code, int32 param0, int32 param1)
     {
      switch(code)
      {
      case MR_MOUSE_DOWN:
      Mggq_a_PointerPressed(param0, param1);
      break;
      case MR_MOUSE_UP:
      Mggq_a_PointerReleased(param0, param1);
      break;
      case MR_KEY_PRESS:
      Mggq_a_Keypressed(param0);
      break;
      case MR_KEY_RELEASE:
      Mggq_a_KeyRelease(param0);
      break;
      default:
      break;
      }
     }
